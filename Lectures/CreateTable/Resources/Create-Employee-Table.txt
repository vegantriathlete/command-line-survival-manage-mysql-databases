CREATE TABLE employee ( first_name CHAR(64) NOT NULL COMMENT 'First Name', last_name CHAR(64) NOT NULL COMMENT 'Last Name', id INTEGER AUTO_INCREMENT PRIMARY KEY COMMENT 'Employee ID', department ENUM( 'Accounting', 'IT', 'Manufacturing', 'Sales') NOT NULL COMMENT 'Department', phone CHAR(16) COMMENT 'Phone Number', hired DATE NOT NULL COMMENT 'Hired Date', termination DATE COMMENT 'Terminated Date');
